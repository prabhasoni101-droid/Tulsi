```
import React from 'react';
import { BrowserRouter as Router, Routes, Route, Navigate } from 'react-router-dom';
import { AuthProvider, useAuth } from './context/AuthContext';
import { AnimatePresence } from 'motion/react';

// Views (Placeholder imports - will create these next)
import Login from './views/Login';
import Dashboard from './views/Dashboard';
import EventDetail from './views/EventDetail';
import DatabaseManagement from './views/DatabaseManagement';
import AttendanceSheet from './views/AttendanceSheet';

const ProtectedRoute: React.FC<{ children: React.ReactNode; minRole?: 'USER' | 'MENTOR' | 'OWNER' }> = ({ children, minRole = 'USER' }) => {
  const { user, profile, loading } = useAuth();
  
  if (loading) return <div className="h-screen w-screen flex items-center justify-center font-serif text-2xl text-saffron animate-pulse">Radhe Radhe...</div>;
  if (!user) return <Navigate to="/login" />;
  
  if (minRole === 'OWNER' && profile?.role !== 'OWNER') return <Navigate to="/" />;
  if (minRole === 'MENTOR' && profile?.role === 'USER') return <Navigate to="/" />;
  
  return <>{children}</>;
};

const AppContent = () => {
  return (
    <Router>
      <AnimatePresence mode="wait">
        <Routes>
          <Route path="/login" element={<Login />} />
          <Route path="/" element={
            <ProtectedRoute>
              <Dashboard />
            </ProtectedRoute>
          } />
          <Route path="/events/:id" element={
            <ProtectedRoute>
              <EventDetail />
            </ProtectedRoute>
          } />
          <Route path="/database" element={
            <ProtectedRoute minRole="MENTOR">
              <DatabaseManagement />
            </ProtectedRoute>
          } />
          <Route path="/attendance" element={
            <ProtectedRoute minRole="OWNER">
              <AttendanceSheet />
            </ProtectedRoute>
          } />
          <Route path="*" element={<Navigate to="/" />} />
        </Routes>
      </AnimatePresence>
    </Router>
  );
};

export default function App() {
  return (
    <AuthProvider>
      <AppContent />
    </AuthProvider>
  );
}
```

```
import React, { createContext, useContext, useEffect, useState } from 'react';
import { User, onAuthStateChanged } from 'firebase/auth';
import { doc, getDoc, setDoc } from 'firebase/firestore';
import { auth, db } from '../services/firebase';
import { UserProfile, UserRole } from '../types';

interface AuthContextType {
  user: User | null;
  profile: UserProfile | null;
  loading: boolean;
  isAdmin: boolean;
  isMentor: boolean;
  isOwner: boolean;
  signOut: () => Promise<void>;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<User | null>(null);
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, async (firebaseUser) => {
      try {
        setLoading(true);
        if (firebaseUser) {
          setUser(firebaseUser);
          // Fetch or create profile
          const profileRef = doc(db, 'users', firebaseUser.uid);
          const profileSnap = await getDoc(profileRef);
          
          if (profileSnap.exists()) {
            const data = profileSnap.data();
            let roleToSet = data.role;
            let templeIdToSet = data.templeId;
            let needsUpdate = false;
            
            // Auto-promote the main admin email if their role is incorrect
            if (firebaseUser.email === 'prabhasoni101@gmail.com' && roleToSet !== 'OWNER') {
              roleToSet = 'OWNER';
              needsUpdate = true;
            }

            if (roleToSet === 'OWNER' && !templeIdToSet) {
              templeIdToSet = firebaseUser.uid;
              needsUpdate = true;
            }

            if (needsUpdate) {
               await setDoc(profileRef, { role: roleToSet, templeId: templeIdToSet }, { merge: true });
            }
            
            setProfile({ uid: firebaseUser.uid, ...data, role: roleToSet, templeId: templeIdToSet } as UserProfile);
          } else {
            // Default role for new signups
            // Since we pre-create profiles for USER/MENTOR in OwnerDashboard before they log in,
            // anyone hitting this block is signing in via Google logic in Login tab (Owner Access).
            const role: UserRole = 'OWNER';
            const newProfile: UserProfile = {
              uid: firebaseUser.uid,
              displayName: firebaseUser.displayName || 'Owner',
              email: firebaseUser.email || '',
              role,
              templeId: firebaseUser.uid,
            };
            await setDoc(profileRef, newProfile);
            setProfile(newProfile);
          }
        } else {
          setUser(null);
          setProfile(null);
        }
      } catch (error) {
        console.error("Auth error:", error);
        setUser(null);
        setProfile(null);
      } finally {
        setLoading(false);
      }
    });

    return unsubscribe;
  }, []);

  const signOut = () => auth.signOut();

  const isOwner = profile?.role === 'OWNER';
  const isMentor = profile?.role === 'MENTOR' || isOwner;
  const isAdmin = isMentor; // Admin is Mentor or Owner

  return (
    <AuthContext.Provider value={{ 
      user, 
      profile, 
      loading, 
      isAdmin, 
      isMentor, 
      isOwner, 
      signOut 
    }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (context === undefined) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
};
```

```
    const result = await signInWithPopup(auth, googleProvider);

    console.error('Error signing in with Google', error);

export async function loginWithUserId(userId: string, password: string) {
  // Map simple ID to internal email format
  const email = `${userId.toLowerCase()}@iskcon.app`;
  return signInWithEmailAndPassword(auth, email, password);
}

export async function registerSevak(userId: string, password: string) {
  const email = `${userId.toLowerCase()}@iskcon.app`;
  
  // Use a secondary app to avoid overwriting the current user's session
  // We use inline require or re-initialize to avoid global scope issues
  const secondaryAppName = 'SecondaryAppForRegistration';
  
  // We use getApps to check if it's already initialized
  let secondaryApp;
  const existingApps = getApps();
  const found = existingApps.find((a: any) => a.name === secondaryAppName);
  if (found) {
    secondaryApp = found;
  } else {
    secondaryApp = initializeApp(firebaseConfig, secondaryAppName);
  }
  
  const secondaryAuth = getAuth(secondaryApp);
  const creds = await createUserWithEmailAndPassword(secondaryAuth, email, password);
  await secondaryAuth.signOut(); // immediately sign out of secondary
  return creds;
}

export interface FirestoreErrorInfo {
  error: string;
  operationType: 'create' | 'update' | 'delete' | 'list' | 'get' | 'write';
  path: string | null;
  authInfo: {
    userId: string;
    email: string;
    emailVerified: boolean;
    isAnonymous: boolean;
    providerInfo: any[];
  }
}

export function handleFirestoreError(error: any, operation: any, path: string | null = null): never {
  const authInfo = auth.currentUser ? {
    userId: auth.currentUser.uid,
    email: auth.currentUser.email || '',
    emailVerified: auth.currentUser.emailVerified,
    isAnonymous: auth.currentUser.isAnonymous,
    providerInfo: auth.currentUser.providerData
  } : {
    userId: 'anonymous',
    email: '',
    emailVerified: false,
    isAnonymous: true,
    providerInfo: []
  };

  const errorInfo: FirestoreErrorInfo = {
    error: error.message || 'Unknown Firestore error',
    operationType: operation,
    path,
    authInfo
  };

  const errorString = JSON.stringify(errorInfo);
  console.error('Firestore Error:', errorString);
  throw new Error(errorString);
}
```

```
import React, { useState, useEffect } from 'react';
import { db } from '../services/firebase';
import { collection, query, onSnapshot, doc, setDoc, getDocs, updateDoc, increment, writeBatch } from 'firebase/firestore';
import { Devotee, Event, Attendance } from '../types';
import Layout from '../components/Layout';
import { 
  FileUp, 
  Search, 
  Check, 
  X, 
  Calendar,
  Filter,
  CheckCircle2
} from 'lucide-react';
import { cn } from '../lib/utils';
import { useAuth } from '../context/AuthContext';

const AttendanceSheet = () => {
  const [events, setEvents] = useState<Event[]>([]);
  const [selectedEventId, setSelectedEventId] = useState<string>('');
  const [devotees, setDevotees] = useState<Devotee[]>([]);
  const [attendance, setAttendance] = useState<Record<string, boolean>>({});
  const [searchTerm, setSearchTerm] = useState('');
  const { profile } = useAuth();

  useEffect(() => {
    const unsubEvents = onSnapshot(query(collection(db, 'events')), (snap) => {
      const eventData = snap.docs.map(doc => ({ id: doc.id, ...doc.data() } as Event));
      setEvents(eventData);
      if (!selectedEventId && eventData.length > 0) setSelectedEventId(eventData[0].id!);
    });

    const unsubDevotees = onSnapshot(query(collection(db, 'devotees')), (snap) => {
      setDevotees(snap.docs.map(doc => ({ id: doc.id, ...doc.data() } as Devotee)));
    });

    return () => {
      unsubEvents();
      unsubDevotees();
    };
  }, [selectedEventId]);

  useEffect(() => {
    if (!selectedEventId) return;
    const unsubAttendance = onSnapshot(collection(db, `events/${selectedEventId}/attendance`), (snap) => {
      const data: Record<string, boolean> = {};
      snap.docs.forEach(doc => {
        data[doc.id] = (doc.data() as Attendance).present;
      });
      setAttendance(data);
    });
    return unsubAttendance;
  }, [selectedEventId]);

  const toggleAttendance = async (devoteeId: string) => {
    if (!selectedEventId) return;
    const isPresent = attendance[devoteeId];
    const docRef = doc(db, `events/${selectedEventId}/attendance`, devoteeId);
    
    // Update attendance record
    await setDoc(docRef, {
      devoteeId,
      present: !isPresent,
      markedAt: new Date().toISOString(),
      markedBy: profile?.uid
    });

    // Update global attendance count for devotee
    await updateDoc(doc(db, 'devotees', devoteeId), {
      attendanceCount: increment(!isPresent ? 1 : -1)
    });
  };

  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file || !selectedEventId) return;

    const reader = new FileReader();
    reader.onload = async (event) => {
      const text = event.target?.result as string;
      const lines = text.split('\n');
      const contacts = lines.map(line => line.trim()).filter(line => line.length > 0);
      
      const batch = writeBatch(db);
      let count = 0;

      for (const contact of contacts) {
        const found = devotees.find(d => d.contact === contact);
        if (found && !attendance[found.id!]) {
          const docRef = doc(db, `events/${selectedEventId}/attendance`, found.id!);
          batch.set(docRef, {
            devoteeId: found.id!,
            present: true,
            markedAt: new Date().toISOString(),
            markedBy: profile?.uid
          });
          batch.update(doc(db, 'devotees', found.id!), {
            attendanceCount: increment(1)
          });
          count++;
        }
      }

      await batch.commit();
      alert(`Successfully marked ${count} devotees as present!`);
    };
    reader.readAsText(file);
  };

  const filteredDevotees = devotees.filter(d => 
    d.name.toLowerCase().includes(searchTerm.toLowerCase()) || 
    d.contact.includes(searchTerm)
  );

  return (
    <Layout>
      <div className="space-y-6">
        <div className="flex flex-col md:flex-row md:items-center justify-between gap-4">
          <div>
            <h1 className="text-4xl text-saffron-dark font-serif font-bold">Attendance Vault</h1>
            <p className="text-stone-500">Record digital attendance through sheets or manual touch.</p>
          </div>
          
          <div className="flex gap-2">
            <label className="btn-secondary flex items-center gap-2 cursor-pointer">
              <FileUp size={18} /> Upload List
              <input type="file" accept=".txt,.csv" className="hidden" onChange={handleFileUpload} />
            </label>
            <select 
              value={selectedEventId} 
              onChange={e => setSelectedEventId(e.target.value)}
              className="bg-white border border-stone-200 rounded-xl px-4 py-2 text-stone-700 font-medium focus:border-saffron outline-none"
            >
              {events.map(e => <option key={e.id} value={e.id}>{e.title}</option>)}
            </select>
          </div>
        </div>

        <div className="iskcon-glass rounded-3xl p-6 space-y-4">
          <div className="flex justify-between items-center bg-stone-50/50 p-4 rounded-2xl">
            <div className="relative flex-1 max-w-md">
              <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-stone-400" size={18} />
              <input 
                type="text" 
                placeholder="Find devotee..."
                className="w-full pl-12 pr-4 py-2 rounded-xl border border-stone-200 outline-none focus:border-saffron bg-white"
                value={searchTerm}
                onChange={e => setSearchTerm(e.target.value)}
              />
            </div>
            <div className="flex items-center gap-6 text-sm font-bold ml-4">
              <div className="text-green-600 flex items-center gap-2">
                <div className="w-2 h-2 bg-green-500 rounded-full" />
                Present: {Object.values(attendance).filter(v => v).length}
              </div>
              <div className="text-stone-400 flex items-center gap-2">
                <div className="w-2 h-2 bg-stone-300 rounded-full" />
                Total: {devotees.length}
              </div>
            </div>
          </div>

          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-3">
            {filteredDevotees.map(d => {
              const isPresent = attendance[d.id!];
              return (
                <button
                  key={d.id}
                  onClick={() => toggleAttendance(d.id!)}
                  className={cn(
                    "flex items-center justify-between p-4 rounded-2xl border transition-all duration-300 group",
                    isPresent 
                      ? "bg-green-50 border-green-200 shadow-sm" 
                      : "bg-white border-stone-100 hover:border-saffron/30"
                  )}
                >
                  <div className="flex items-center gap-3">
                    <div className={cn(
                      "w-10 h-10 rounded-full flex items-center justify-center font-bold text-sm",
                      isPresent ? "bg-green-600 text-white" : "bg-stone-50 text-stone-400"
                    )}>
                      {isPresent ? <Check size={20} /> : d.name[0]}
                    </div>
                    <div className="text-left">
                      <p className={cn("font-bold text-sm", isPresent ? "text-green-800" : "text-stone-700")}>{d.name}</p>
                      <p className="text-[10px] text-stone-400 font-mono italic">{d.contact}</p>
                    </div>
                  </div>
                  <div className={cn(
                    "w-6 h-6 rounded-lg flex items-center justify-center transition-all",
                    isPresent ? "bg-green-100 text-green-600" : "bg-stone-50 text-stone-200 group-hover:text-stone-400"
                  )}>
                    {isPresent ? <CheckCircle2 size={14} /> : <X size={14} />}
                  </div>
                </button>
              );
            })}
          </div>
        </div>
      </div>
    </Layout>
  );
};

export default AttendanceSheet;
```

```
    });
```

```
      const allUsers = snapshot.docs.map(doc => ({ uid: doc.id, ...doc.data() } as UserProfile));

          const canvas = document.createElement('canvas');
          const MAX_WIDTH = 800;
          let width = img.width;
          let height = img.height;
          
          if (width > MAX_WIDTH) {
            height = Math.round((height * MAX_WIDTH) / width);
            width = MAX_WIDTH;
          }
          
          canvas.width = width;
          canvas.height = height;
          const ctx = canvas.getContext('2d');
          ctx?.drawImage(img, 0, 0, width, height);

                  Register Another
```

```
import React, { useState, useEffect } from 'react';
import { db } from '../../services/firebase';
import { collection, query, where, onSnapshot, addDoc, serverTimestamp } from 'firebase/firestore';
import { Event, Devotee } from '../../types';
import { useAuth } from '../../context/AuthContext';
import { motion } from 'motion/react';
import { Calendar, UserPlus, Phone, ChevronRight, CheckCircle2 } from 'lucide-react';
import { useNavigate } from 'react-router-dom';
import { cn } from '../../lib/utils';

const UserDashboard = () => {
  const [activeTab, setActiveTab] = useState<'events' | 'facilitation'>('events');
  const [events, setEvents] = useState<Event[]>([]);
  const [newDevotee, setNewDevotee] = useState({ name: '', contact: '', mentor: '', chanting: '' });
  const [recentAdditions, setRecentAdditions] = useState<Devotee[]>([]);
  const { profile } = useAuth();
  const navigate = useNavigate();

  useEffect(() => {
    // Only fetch public events
    const qE = query(collection(db, 'events'), where('isPublic', '==', true));
    const unsubscribeE = onSnapshot(qE, (snapshot) => {
      setEvents(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() } as Event)));
    });

    let unsubscribeD = () => {};
    // Recent facilitation by this user
    if (profile?.uid) {
      const qD = query(collection(db, 'devotees'), where('facilitatorId', '==', profile.uid));
      unsubscribeD = onSnapshot(qD, (snapshot) => {
        setRecentAdditions(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() } as Devotee)));
      });
    }

    return () => {
      unsubscribeE();
      unsubscribeD();
    };
  }, [profile]);

  const handleAddDevotee = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!newDevotee.name || !newDevotee.contact) return;

    try {
      await addDoc(collection(db, 'devotees'), {
        ...newDevotee,
        facilitatorId: profile?.uid,
        facilitatorName: profile?.displayName || 'Anonymous',
        attendanceCount: 0,
        createdAt: serverTimestamp(),
      });
      setNewDevotee({ name: '', contact: '', mentor: '', chanting: '' });
      alert('Devotee added successfully!');
    } catch (error) {
      console.error(error);
      alert('Error adding devotee. Duplicate entry might be restricted.');
    }
  };

  return (
    <div className="space-y-8">
      <div className="flex flex-col md:flex-row md:items-center justify-between gap-4">
        <div>
          <h2 className="text-3xl font-serif text-saffron-dark">Sevak Dashboard</h2>
          <p className="text-stone-500 italic font-serif">"Selfless service is the highest worship."</p>
        </div>
        <div className="flex bg-stone-100 p-1 rounded-2xl border border-stone-200">
          <button 
            onClick={() => setActiveTab('events')}
            className={cn("px-6 py-2 rounded-xl transition-all font-medium", activeTab === 'events' ? "bg-white shadow text-saffron-dark" : "text-stone-500")}
          >
            Events
          </button>
          <button 
            onClick={() => setActiveTab('facilitation')}
            className={cn("px-6 py-2 rounded-xl transition-all font-medium", activeTab === 'facilitation' ? "bg-white shadow text-saffron-dark" : "text-stone-500")}
          >
            Facilitation
          </button>
        </div>
      </div>

      {activeTab === 'events' && (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {events.length === 0 ? (
            <div className="col-span-full py-20 text-center text-stone-400 font-serif">
              <Calendar size={64} className="mx-auto opacity-10 mb-4" />
              <p>No active events found. Please wait for the administrator to publish events.</p>
            </div>
          ) : (
            events.map((event) => (
              <motion.div 
                key={event.id}
                whileHover={{ y: -5 }}
                className="iskcon-glass rounded-3xl p-6 space-y-4 hover:border-saffron/30 transition-all cursor-pointer"
                onClick={() => navigate(`/events/${event.id}`)}
              >
                <div className="flex justify-between items-start">
                  <div className="bg-saffron/10 p-3 rounded-2xl text-saffron">
                    <Calendar size={24} />
                  </div>
                  <ChevronRight size={20} className="text-stone-300" />
                </div>
                <div>
                  <h3 className="text-xl font-bold text-stone-800">{event.title}</h3>
                  <p className="text-sm text-stone-500 mt-1">{new Date(event.date).toLocaleDateString(undefined, { dateStyle: 'full' })}</p>
                </div>
                <div className="bg-saffron/5 rounded-2xl p-4 text-sm text-stone-600">
                  <p className="line-clamp-2 italic font-serif leading-relaxed opacity-80">
                    {event.description || 'Join us for this blissful celebration at the temple.'}
                  </p>
                </div>
                <div className="pt-2 flex items-center justify-between text-xs font-bold text-saffron uppercase">
                  <span>View My List</span>
                  <span>Open Seva &rarr;</span>
                </div>
              </motion.div>
            ))
          )}
        </div>
      )}

      {activeTab === 'facilitation' && (
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
          <div className="lg:col-span-2 space-y-6">
            <div className="iskcon-glass rounded-3xl p-8 space-y-6">
              <div className="flex items-center gap-4 border-b border-stone-100 pb-4">
                <div className="bg-gold/20 p-2 rounded-lg text-saffron-dark"><UserPlus size={24} /></div>
                <h3 className="text-2xl font-serif">Add New Devotee</h3>
              </div>
              
              <form onSubmit={handleAddDevotee} className="grid grid-cols-1 md:grid-cols-2 gap-6">
                <div className="space-y-1">
                  <label className="text-xs font-bold text-stone-500 uppercase">Full Name</label>
                  <input 
                    type="text" required
                    className="w-full px-4 py-3 rounded-xl border border-stone-100 bg-stone-50/50 focus:bg-white focus:border-saffron outline-none transition-all"
                    value={newDevotee.name}
                    onChange={e => setNewDevotee({...newDevotee, name: e.target.value})}
                    placeholder="Enter name"
                  />
                </div>
                <div className="space-y-1">
                  <label className="text-xs font-bold text-stone-500 uppercase">Contact Number</label>
                  <input 
                    type="tel" required
                    className="w-full px-4 py-3 rounded-xl border border-stone-100 bg-stone-50/50 focus:bg-white focus:border-saffron outline-none transition-all"
                    value={newDevotee.contact}
                    onChange={e => setNewDevotee({...newDevotee, contact: e.target.value})}
                    placeholder="Phone number"
                  />
                </div>
                <div className="space-y-1">
                  <label className="text-xs font-bold text-stone-500 uppercase">Mentor Name</label>
                  <input 
                    type="text"
                    className="w-full px-4 py-3 rounded-xl border border-stone-100 bg-stone-50/50 focus:bg-white focus:border-saffron outline-none transition-all"
                    value={newDevotee.mentor}
                    onChange={e => setNewDevotee({...newDevotee, mentor: e.target.value})}
                    placeholder="Reference mentor"
                  />
                </div>
                <div className="space-y-1">
                  <label className="text-xs font-bold text-stone-500 uppercase">Chanting (Rounds)</label>
                  <input 
                    type="number"
                    className="w-full px-4 py-3 rounded-xl border border-stone-100 bg-stone-50/50 focus:bg-white focus:border-saffron outline-none transition-all"
                    value={newDevotee.chanting}
                    onChange={e => setNewDevotee({...newDevotee, chanting: e.target.value})}
                    placeholder="How many rounds?"
                  />
                </div>
                <button type="submit" className="md:col-span-2 btn-primary py-4 text-lg mt-2">Save Devotee Record</button>
              </form>
            </div>
          </div>

          <div className="space-y-4">
            <h4 className="text-sm font-bold text-stone-400 uppercase tracking-widest pl-2">Recent Additions</h4>
            <div className="space-y-3">
              {recentAdditions.map((d) => (
                <div key={d.id} className="bg-white border border-stone-100 rounded-2xl p-4 flex items-center justify-between">
                  <div>
                    <p className="font-bold text-stone-800">{d.name}</p>
                    <p className="text-xs text-stone-500">{d.contact}</p>
                  </div>
                  <div className="bg-green-100 text-green-600 p-2 rounded-full">
                    <CheckCircle2 size={16} />
                  </div>
                </div>
              ))}
              {recentAdditions.length === 0 && (
                <p className="text-sm text-stone-400 italic text-center py-10 bg-stone-50/50 rounded-2xl border border-dashed border-stone-200">
                  No devotees added yet.
                </p>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default UserDashboard;
```
